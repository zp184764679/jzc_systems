# 服务端口分配
# ─────────────────────────────────────
# portal-backend     3002   Flask
# hr-backend         8003   Flask
# account-backend    8004   Flask
# quotation-backend  8001   FastAPI (uvicorn)
# caigou-backend     5001   Flask
# shm-backend        8006   Flask


# Nginx 路由
# ─────────────────────────────────────
# /              → Portal/dist
# /api/          → portal-backend:3002
# /hr/           → HR/frontend/dist
# /hr/api/       → hr-backend:8003
# /account/      → account/frontend/dist
# /account/api/  → account-backend:8004
# /quotation/    → 报价/frontend/dist
# /quotation/api/→ quotation-backend:8001
# /caigou/       → 采购/frontend/dist
# /caigou/api/   → caigou-backend:5001
# /shm/          → SHM/frontend/dist
# /shm/api/      → shm-backend:8006
